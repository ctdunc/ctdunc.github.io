     39 function Tex_GetCurrentEnv()
     40 	let i = 0
     41 	while 1
     42 		let env_line = search('^[^%]*\\\%(begin\|end\){', 'bW')
     43 		if env_line == 0
     44 			return ''
     45 		endif
     46 		if match(getline(env_line), '^[^%]*\\begin{') == -1
     47 			let i = i +1
     48 			continue
     49 		else
     50 			if i==0
     51 				let env = matchstr(getline(env_line), '\\begin{\zs.\{-}\ze}')
     52 				exe echo env
     53 				return env
     54 			else
     55 				let i = i-1
     56 				continue
     57 			endif
     58 		endif
     59 	endwhile
     60 endfunction
