+function Tex_GetCurrentEnv()
+	let i = 0
+	while 1
+		let env_line = search('^[^%]*\\\%(begin\|end\){', 'bW')
+		if env_line == 0
+			return ''
+		endif
+		if match(getline(env_line), '^[^%]*\\begin{') == -1
+			let i = i +1
+			continue
+		else
+			if i==0
+				let env = matchstr(getline(env_line), '\\begin{\zs.\{-}\ze}')
+				exe echo env
+				return env
+			else
+				let i = i-1
+				continue
+			endif
+		endif
+	endwhile
+endfunction
